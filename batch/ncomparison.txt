displaycode off
displayresult off
string resultdir dataset data file1 file2 test comptype
integer i j k numberofruns
real count
setprecision 3
twotailed
= numberofruns 100
= resultdir ../results/multivariatetest/real
= dataset spambase
for i 1 4
  switch i
    case 1
      = comptype error
      = test cvttest
    case 2
      = comptype confusion
      = test hotellingt
    case 3
      = comptype roc
      = test hotellingt
    case 4
      = comptype pr
      = test hotellingt
  endswitch
  output %resultdir%/%comptype%-%dataset%.txt
  for j 1 10
    = data %dataset%%j%
    = count 0
    for k 1 %numberofruns%
      = file1 %resultdir%/%dataset%/%k%-%comptype%.txt
      = file2 %resultdir%/%data%/%k%-%comptype%.txt
      %test% %file1% %file2%

        ++ count
      endif
    endfor k
    write %j%_%count/numberofruns%
    writescreen %i%_%j%
  endfor j
endfor i
